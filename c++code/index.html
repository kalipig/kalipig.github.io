inline int read()
{
    int x=0,f=1;
    char ch=getchar();
    while(ch<'0'||ch>'9')
    {
        if(ch=='-')
            f=-1;
        ch=getchar();
    }
    while(ch>='0' && ch<='9')
        x=x*10+ch-'0',ch=getchar();
    return x*f;
}

vector<int> add(const vector<int>& num1, const vector<int>& num2) {
    int carry = 0;
    vector<int> result;
    
    int len1 = num1.size(), len2 = num2.size();
    int len = max(len1, len2);
    
    for (int i = 0; i < len; ++i) {
        int sum = carry;
        if (i < len1) {
            sum += num1[i];
        }
        if (i < len2) {
            sum += num2[i];
        }
        
        carry = sum / 10;
        result.push_back(sum % 10);
    }
    
    if (carry) {
        result.push_back(carry);
    }
    
    return result;
}

vector<int> subtract(const vector<int>& num1, const vector<int>& num2) {
    vector<int> result;
    int borrow = 0;
    
    int len1 = num1.size(), len2 = num2.size();
    int len = max(len1, len2);
    
    for (int i = 0; i < len; ++i) {
        int diff = borrow;
        if (i < len1) {
            diff += num1[i];
        }
        if (i < len2) {
            diff -= num2[i];
        }
        
        if (diff < 0) {
            diff += 10;
            borrow = -1;
        } else {
            borrow = 0;
        }
        
        result.push_back(diff);
    }
    
    while (result.size() > 1 && result.back() == 0) {
        result.pop_back();
    }
    
    return result;
}
vector<int> intToVector(int num) {
    vector<int> result;
    while (num > 0) {
        result.push_back(num % 10);
        num /= 10;
    }
    if (result.empty()) {
        result.push_back(0); // 处理 0 的情况
    }
    return result;
}
